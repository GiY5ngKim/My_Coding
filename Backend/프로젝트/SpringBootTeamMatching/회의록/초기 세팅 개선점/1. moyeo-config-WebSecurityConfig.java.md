## ✅ 줄별 역할/기능 해설

```java
package com.catholic.moyeo.config;
// 이 클래스가 속한 패키지 선언. 보통 설정(config) 클래스를 모아두는 네임스페이스.

import com.catholic.moyeo.security.JwtAuthenticationFilter;
// 요청마다 JWT를 검사하여 인증 정보를 SecurityContext에 세팅하는 커스텀 필터(추정).

import com.catholic.moyeo.security.OAuth2SuccessHandler;
// OAuth2 로그인 성공 직후 실행되는 성공 핸들러(예: JWT 발급, 리다이렉트/응답 처리 등).

import lombok.RequiredArgsConstructor;
// final 필드들을 파라미터로 받는 생성자를 Lombok이 자동 생성(생성자 주입).

import org.springframework.context.annotation.Bean;
// @Bean: 메서드 반환 객체를 스프링 컨테이너에 빈으로 등록.

import org.springframework.context.annotation.Configuration;
// @Configuration: 이 클래스가 스프링 설정 클래스임을 선언.

import org.springframework.http.HttpStatus;
// HTTP 상태코드(401, 403 등) 사용을 위한 enum.

import org.springframework.security.config.annotation.web.builders.HttpSecurity;
// Spring Security 설정을 구성하는 DSL 빌더.

import org.springframework.security.config.http.SessionCreationPolicy;
// 세션 생성 정책(STATELESS 등) 지정에 사용.

import org.springframework.security.web.SecurityFilterChain;
// Spring Security 필터 체인을 빈으로 등록하기 위한 타입.

import org.springframework.security.web.authentication.HttpStatusEntryPoint;
// 인증 실패 시 특정 HTTP 상태코드를 반환하는 AuthenticationEntryPoint 구현체.

import org.springframework.security.web.authentication.UsernamePasswordAuthenticationFilter;
// 폼 로그인(아이디/비번) 처리 필터. JWT 필터를 이 필터보다 앞에 두기 위한 기준점.

import org.springframework.web.cors.CorsConfiguration;
// CORS 허용 Origin/Method/Header 등 정책을 담는 설정 객체.

import org.springframework.web.cors.CorsConfigurationSource;
// 요청에 대해 어떤 CORS 설정을 적용할지 제공하는 인터페이스.

import org.springframework.web.cors.UrlBasedCorsConfigurationSource;
// URL 패턴별 CORS 설정을 매핑할 수 있는 구현체.

/**
 * Spring Security 설정
 *
 * 구성:
 * - OAuth2 로그인
 * - JWT 인증 필터
 * - 인증 정책 설정
 */
@Configuration
// 이 클래스가 스프링 설정 클래스임을 선언. (여기서 만든 빈들이 컨테이너에 등록됨)

@RequiredArgsConstructor
// final 필드(oAuth2SuccessHandler, jwtAuthenticationFilter)를 주입받는 생성자를 자동 생성.
public class WebSecurityConfig {

    private final OAuth2SuccessHandler oAuth2SuccessHandler;
    // OAuth2 로그인 성공 시 후처리(토큰 발급/응답 처리)를 담당하는 핸들러.

    private final JwtAuthenticationFilter jwtAuthenticationFilter;
    // 매 요청 JWT 검증을 수행하는 커스텀 필터.

    @Bean
    // SecurityFilterChain을 스프링 빈으로 등록. Spring Security가 이 설정을 적용하게 됨.
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {

        http
                .csrf(csrf -> csrf.disable())
                // CSRF 보호 기능 비활성화.
                // JWT 헤더 기반 stateless API에서는 보통 비활성화(쿠키 세션이면 위험).

                .cors(cors -> cors.configurationSource(corsConfigurationSource()))
                // CORS 설정 활성화 + 아래 corsConfigurationSource()에서 정의한 정책 적용.

                .sessionManagement(sm -> sm.sessionCreationPolicy(SessionCreationPolicy.STATELESS))
                // 세션을 사용/생성하지 않는 정책 설정(STATELESS).
                // 서버가 로그인 상태를 세션으로 저장하지 않고 매 요청 토큰으로 인증한다는 의미.

                .formLogin(form -> form.disable())
                // 기본 폼 로그인(HTML 로그인 페이지 기반) 비활성화.

                .httpBasic(basic -> basic.disable())
                // HTTP Basic 인증(Authorization: Basic ...) 비활성화.



                //  401/403 응답 고정
                .exceptionHandling(ex -> ex
                        // 인증 안됨 -> 401
                        .authenticationEntryPoint(new HttpStatusEntryPoint(HttpStatus.UNAUTHORIZED))
                        // 인증 정보가 없거나/유효하지 않을 때 401(Unauthorized)로 응답하도록 설정.

                        // 권한 없음 -> 403
                        .accessDeniedHandler((req, res, e) -> res.sendError(HttpStatus.FORBIDDEN.value()))
                        // 인증은 됐지만 권한이 부족할 때 403(Forbidden)으로 응답하도록 설정.
                )


                // URL 접근 정책
                .authorizeHttpRequests(auth -> auth
                        .requestMatchers(
                                "/",
                                "/oauth2/**",
                                "/login/**",
                                "/oauth/**",
                                "/oauth/callback"
                        ).permitAll()
                        // 위 경로들은 인증 없이 접근 허용(공개 엔드포인트).

                        .anyRequest().authenticated()
                        // 나머지 모든 요청은 인증 필요.
                )

                // OAuth2 로그인 설정
                .oauth2Login(oauth -> oauth
                        .successHandler(oAuth2SuccessHandler)
                        // OAuth2 로그인 성공 시 커스텀 성공 핸들러 실행.
                )

                // JWT 필터 등록
                .addFilterBefore(
                        jwtAuthenticationFilter,
                        UsernamePasswordAuthenticationFilter.class
                );
                // UsernamePasswordAuthenticationFilter(폼로그인 처리)보다 앞에 JWT 필터를 배치.
                // 즉, 요청이 들어오면 먼저 JWT로 인증을 시도하게 됨.

        return http.build();
        // 위 설정을 기반으로 SecurityFilterChain을 생성하여 반환.
    }


    @Bean
    // CORS 설정 소스를 스프링 빈으로 등록.
    public CorsConfigurationSource corsConfigurationSource() {
        CorsConfiguration config = new CorsConfiguration();
        // CORS 정책을 담을 설정 객체 생성.

        config.addAllowedOriginPattern("*");
        // 모든 Origin(출처)에서 오는 요청을 허용(패턴 기반).

        config.addAllowedMethod("*");
        // 모든 HTTP 메서드(GET/POST/PUT/DELETE 등) 허용.

        config.addAllowedHeader("*");
        // 모든 헤더 허용(Authorization 같은 헤더 포함).

        config.setAllowCredentials(true);
        // 쿠키/인증정보 포함 요청을 허용.
        // (브라우저 정책상 Origin '*'와 함께 쓰면 이슈가 생길 수 있음)

        UrlBasedCorsConfigurationSource source = new UrlBasedCorsConfigurationSource();
        // URL 패턴에 따라 CORS 설정을 매핑하는 소스 객체 생성.

        source.registerCorsConfiguration("/**", config);
        // 모든 경로("/**")에 대해 위 CORS 정책 적용.

        return source;
        // Security가 이 CORS 설정 소스를 사용하도록 반환.
    }
}
```
## 🔴 P0 — 반드시 수정 (구조 충돌 / 장애 가능성 / 보안 리스크)

---

### 1️⃣ 로그인 엔드포인트 permitAll 누락 가능성

#### 📌 현재 코드

```java
.requestMatchers(
    "/",
    "/oauth2/**",
    "/login/**",
    "/oauth/**",
    "/oauth/callback"
).permitAll()
```

#### ❌ 문제 이유

- 명세상 로그인 API는 `/api/auth/oauth`
    
- permitAll에 포함되어 있지 않으면 로그인 자체가 401로 막힐 수 있음
    
- 실제 장애로 이어질 가능성 높음
    

#### ✅ 수정안

```java
.authorizeHttpRequests(auth -> auth
    .requestMatchers(
        "/api/auth/oauth",
        "/api/auth/logout"
    ).permitAll()
    .anyRequest().authenticated()
)
```

---

### 2️⃣ oauth2Login 혼합 구조 제거

#### 📌 현재 코드

```java
.oauth2Login(oauth -> oauth
    .successHandler(oAuth2SuccessHandler)
)
```

#### ❌ 문제 이유

- 현재 로그인 방식은 idToken 전달 방식
    
- oauth2Login은 리다이렉트 기반 OAuth 플로우용
    
- 두 구조 혼합 시 세션/상태 관리 충돌 가능
    
- 유지보수 복잡성 증가
    

#### ✅ 수정안

해당 블록 **완전 제거**

```java
// .oauth2Login(...) 제거
```

---

### 3️⃣ CORS 설정 위험 조합

#### 📌 현재 코드

```java
config.addAllowedOriginPattern("*");
config.setAllowCredentials(true);
```

#### ❌ 문제 이유

- `*` + credentials=true 조합은 브라우저 정책상 문제 발생 가능
    
- 모든 출처 허용은 보안상 위험
    
- 실서비스에서는 거의 사용하지 않음
    

#### ✅ 수정안 (화이트리스트 방식)

```java
config.setAllowedOrigins(List.of(
    "http://localhost:3000",
    "https://moyeo.com"
));
config.setAllowCredentials(false);
config.setAllowedMethods(List.of("GET","POST","PUT","PATCH","DELETE","OPTIONS"));
config.setAllowedHeaders(List.of("Authorization","Content-Type"));
```

---

### 4️⃣ JWT 필터 통과 조건 점검 필요

#### 📌 문제 가능 지점

```java
.addFilterBefore(jwtAuthenticationFilter, UsernamePasswordAuthenticationFilter.class)
```

#### ❌ 위험 이유

- JWT 필터 구현이 “토큰 없으면 401”이면  
    permitAll 엔드포인트까지 막힘
    

#### ✅ 수정 원칙 (필터 내부)

- 토큰 없음 → 그냥 통과
    
- 토큰 있음 + invalid → 401
    
- 토큰 있음 + valid → SecurityContext 세팅
    

(이건 JwtAuthenticationFilter 파일에서 확정 분석 필요)

## 🟡 P1 — 실무 권장 (졸업프로젝트에서는 선택)

---

### 5️⃣ 401/403 JSON 응답 통일

#### 📌 현재 코드

```java
.authenticationEntryPoint(new HttpStatusEntryPoint(HttpStatus.UNAUTHORIZED))
.accessDeniedHandler((req, res, e) -> res.sendError(HttpStatus.FORBIDDEN.value()))
```

#### 개선 이유

- 프론트에서 에러 처리 통일 필요
    
- 상태코드만 보내는 방식은 확장성 부족
    

#### 추천

CustomAuthenticationEntryPoint / CustomAccessDeniedHandler 작성

---

### 6️⃣ CORS 설정을 application.yml로 분리

#### 개선 이유

- 환경별 도메인 관리 용이
    
- 배포 시 코드 수정 불필요
    

졸업프로젝트에서는 하드코딩해도 무방.
