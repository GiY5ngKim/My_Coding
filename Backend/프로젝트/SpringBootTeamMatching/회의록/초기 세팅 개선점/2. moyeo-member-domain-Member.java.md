## ✅ 줄별 역할/기능 해설

```java
package com.catholic.moyeo.member.domain;
// 이 클래스의 패키지 경로 선언. (논리적 폴더/네임스페이스)

import jakarta.persistence.*;
// JPA 엔티티 매핑 관련 어노테이션(@Entity, @Id, @Column 등)을 사용하기 위해 전체 임포트.

import lombok.Getter;
// Lombok의 @Getter 사용을 위해 임포트. 필드별 getter 메서드를 자동 생성함.

import org.springframework.data.annotation.CreatedDate;
// Spring Data Auditing에서 "생성 시각" 자동 주입용 어노테이션.

import org.springframework.data.annotation.LastModifiedDate;
// Spring Data Auditing에서 "수정 시각" 자동 주입용 어노테이션.

import org.springframework.data.jpa.domain.support.AuditingEntityListener;
// Auditing 이벤트(생성/수정 시각 자동 세팅)를 감지해 처리하는 리스너.

import java.time.LocalDateTime;
// createdAt/updatedAt에 사용할 날짜-시간 타입.

@EntityListeners(AuditingEntityListener.class)
// 이 엔티티에 AuditingEntityListener를 붙여서
// @CreatedDate, @LastModifiedDate가 자동으로 채워지도록 함.
// (전제: 프로젝트에 @EnableJpaAuditing 설정이 켜져 있어야 실제 동작함)

@Getter
// 모든 필드에 대한 getter 메서드를 Lombok이 자동 생성.
// (예: getId(), getEmail() 등)

@Entity
// 이 클래스가 JPA 관리 대상 "엔티티"임을 선언.
// 즉, DB 테이블의 행(row)과 매핑되는 객체가 됨.

@Table(
        name = "app_user",
// 이 엔티티가 매핑될 테이블 이름을 "app_user"로 지정.
// 기본값(클래스명 기반)이 아닌 명시적 테이블명 사용.

        uniqueConstraints = {
                @UniqueConstraint(name = "uk_app_user_email", columnNames = "email"),
// email 컬럼에 유니크 제약조건을 추가.
// 제약 이름: uk_app_user_email
// 효과: 같은 email을 가진 row를 DB에 2개 이상 저장 불가.

                @UniqueConstraint(name = "uk_app_user_provider_sub", columnNames = {"provider", "provider_sub"})
// (provider, provider_sub) 조합에 유니크 제약조건을 추가.
// 제약 이름: uk_app_user_provider_sub
// 효과: 같은 provider에서 같은 provider_sub(예: 구글 sub)로 중복 가입 불가.
        }
)
public class Member {
// 사용자(회원) 엔티티 클래스. DB의 app_user 테이블과 매핑됨.

    @Id
// 이 필드가 PK(Primary Key)임을 선언.

    @GeneratedValue(strategy = GenerationType.IDENTITY)
// PK 생성 전략을 IDENTITY로 지정.
// DB의 auto-increment(예: MySQL AUTO_INCREMENT, PostgreSQL SERIAL/IDENTITY) 방식으로 증가하는 값을 사용.

    @Column(name = "user_id")
// DB 컬럼명을 user_id로 지정. (필드명 id와 DB 컬럼명 불일치 해결)
    private Long id;
// 사용자 PK.

    @Column(nullable = false, length = 20)
// provider 컬럼: NULL 불가, 최대 길이 20.
    private String provider; // google
// OAuth 공급자 문자열 저장 용도(예: "google").
// 주석은 현재 구글만 사용한다는 의미.

    @Column(name = "provider_sub", nullable = false, length = 255)
// DB 컬럼명을 provider_sub로 지정.
// NULL 불가, 최대 길이 255.
    private String providerSub; // 구글 subject
// OAuth 공급자에서 제공하는 subject(sub) 값.
// 구글의 경우 idToken의 "sub" 클레임 값이 여기 들어감.

    @Column(nullable = false, length = 255)
// email 컬럼: NULL 불가, 최대 길이 255.
    private String email;
// 사용자 이메일 저장.

    @Column(name = "email_verified", nullable = false)
// DB 컬럼명을 email_verified로 지정.
// NULL 불가. (boolean은 기본값 false로 들어가긴 하지만, 컬럼 NULL 방지)
    private boolean emailVerified;
// 이메일 검증 여부 저장.
// 구글 idToken의 email_verified 클레임과 매핑되는 값.

    @CreatedDate
// 엔티티가 "처음 저장될 때" 생성 시각을 자동으로 세팅하라는 의미.
// Spring Data JPA Auditing이 활성화되어 있어야 적용됨.

    @Column(name = "created_at", nullable = false, updatable = false)
// DB 컬럼명 created_at.
// NULL 불가.
// updatable=false → UPDATE 쿼리에서 이 컬럼은 변경 대상으로 포함되지 않도록 함.
    private LocalDateTime createdAt;
// 생성 시각.

    @LastModifiedDate
// 엔티티가 저장/수정될 때 "마지막 수정 시각"을 자동으로 세팅하라는 의미.

    @Column(name = "updated_at", nullable = false)
// DB 컬럼명 updated_at.
// NULL 불가.
    private LocalDateTime updatedAt;
// 마지막 수정 시각.



    protected Member() {}
// JPA 스펙상 엔티티는 기본 생성자가 필요.
// 접근을 protected로 해서 외부에서 "무분별한 new Member()"를 막고,
// JPA 프록시/리플렉션 생성만 허용하는 패턴.

    private Member(String provider, String providerSub, String email, boolean emailVerified) {
// 외부에서 직접 생성 못 하도록 private 생성자.
// 정적 팩토리 메서드(createGoogle)를 통해서만 생성하게 유도.

        this.provider = provider;
// provider 필드 초기화.

        this.providerSub = providerSub;
// providerSub 필드 초기화.

        this.email = email;
// email 필드 초기화.

        this.emailVerified = emailVerified;
// emailVerified 필드 초기화.
    }

    public static Member createGoogle(String sub, String email, boolean verified) {
// 구글 OAuth 사용자 생성 전용 정적 팩토리 메서드.
// 호출자는 "provider는 항상 google"이라는 규칙을 몰라도 됨.

        return new Member("google", sub, email, verified);
// 실제 엔티티 객체 생성.
// provider="google"로 고정하고 sub/email/verified는 전달값을 사용.
    }
}
```


# 🔴 P0 — 반드시 점검/수정해야 할 사항 (구조 안정성 / 운영 리스크)
---
## 1️⃣ `email` 유니크 제약의 정책 검토 필요

### 📌 현재 코드

```java
@UniqueConstraint(name = "uk_app_user_email", columnNames = "email")
```

### ❗ 문제 가능성

현재 설계는:

- email 단일 유니크
    
- (provider, provider_sub) 복합 유니크
    

즉:

- 같은 이메일로 다른 provider 가입 불가
    

현재는 구글만 사용 → 문제 없음  
하지만 향후 Kakao/Naver 추가 시:

- 같은 이메일인데 provider가 다르면 가입 불가
    
- 정책 충돌 발생
    

### ✅ 실무 권장 판단

|경우|권장|
|---|---|
|provider 1개만 사용|email unique 유지 가능|
|provider 다중 지원 계획|email unique 제거하고 (provider, provider_sub)만 유지|

📌 졸업 프로젝트라면  
→ 구글만 확정이면 유지 가능

---

## 2️⃣ `provider`를 문자열로 관리하는 문제

### 📌 현재 코드

```java
private String provider; // google
```

### ❗ 문제

- 문자열은 오타 위험
    
- 대소문자 혼합 가능
    
- "google", "Google", "GOOGLE" 문제 발생 가능
    

### ✅ 실무 권장

Enum으로 관리

```java
@Enumerated(EnumType.STRING)
@Column(nullable = false, length = 20)
private AuthProvider provider;
```

```java
public enum AuthProvider {
    GOOGLE
}
```

### 🎯 이유

- 타입 안정성
    
- 오타 방지
    
- 확장성
    

📌 졸업 프로젝트에서도 이건 추천 (P0에 가까운 P1 상위)

---

## 3️⃣ `emailVerified` 신뢰성 검증 필요

### 📌 현재 코드

```java
private boolean emailVerified;
```

### ❗ 문제 가능성

- 이 값은 반드시 Google idToken 검증 결과에서만 세팅되어야 함
    
- 클라이언트에서 전달받은 값 사용하면 위험
    

### ✅ 실무 권장

- idToken 서버 검증 후 claim 기반으로만 세팅
    
- 컨트롤러에서 직접 값 받지 말 것
    

---

## 4️⃣ Auditing 활성화 전제 조건 확인 필요

### 📌 현재 코드

```java
@EntityListeners(AuditingEntityListener.class)
@CreatedDate
@LastModifiedDate
```

### ❗ 전제 조건

프로젝트에 반드시 있어야 함:

```java
@EnableJpaAuditing
```

없으면 createdAt/updatedAt null 발생 → DB insert 오류

📌 이건 실무/졸업프로젝트 공통 P0

---

# 🟡 P1 — 실무 권장 개선 (졸업프로젝트에서는 선택)

---

## 5️⃣ Soft Delete 전략 검토

현재는 물리 삭제 구조.

실무에서는:

```java
private boolean deleted;
```

또는

```java
private LocalDateTime deletedAt;
```

사용자 탈퇴 시 완전 삭제 대신 Soft Delete.

졸업 프로젝트라면 생략 가능.

---

## 6️⃣ email 컬럼 길이 255는 적절

✔ 문제 없음  
실무에서도 일반적으로 254~255 사용

---

## 7️⃣ 생성자 패턴은 적절

```java
protected Member() {}
private Member(...)
public static Member createGoogle(...)
```

✔ JPA 기본 생성자 보호  
✔ 정적 팩토리 메서드 사용  
✔ 외부 new 차단

이 구조는 실무 수준으로 양호

---

## 8️⃣ providerSub 길이 255 적절

구글 sub는 255 이하  
문제 없음

---

# 🔵 P2 — 운영/확장 단계 고려

---

## 9️⃣ createdAt / updatedAt 타입

현재:

```java
LocalDateTime
```

실무에서는:

- `OffsetDateTime` 또는
    
- DB timezone 명확 설정
    

클라우드/다중 리전 환경 아니면 졸업프로젝트에서는 LocalDateTime 충분.

---

## 🔟 Index 명시적 추가 고려

provider_sub 조회가 빈번하면:

```java
@Index(name="idx_provider_sub", columnList="provider_sub")
```

추가 고려 가능.

졸업 프로젝트에선 필수 아님.
